How it works, a simple example:

HTML

      
        <div> id="app">
          {{ message }}
        </div>

JS

      
        var app = new Vue({
          el: '#app',
          data: {
            message: 'Working with Vue!'
          }
        });

Component Systems

register a component in vue

      
        <ol>
          
          
          <todo-item v-for="item in nicksList" v-bind:todo="item">
        </ol>
      
    
      
        Vue.component('todo-item', {
          // The todo-item component accepts a
          // "prop", which is like a custom attribute.
          // This prop is called todo.
          props: ['todo'],
          template: '<li>{{ todo.text }}'
        })
